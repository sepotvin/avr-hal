[−][src]Type Definition arduino_leonardo::I2c

type I2c<M> = I2c<MHz16, M>;

I2C Master on pins D2 (SDA) and D3 (SCL)

Example

For a full example, see examples/leonardo-i2cdetect.rs. In short:

let dp = arduino_leonardo::Peripherals::take().unwrap();

let mut pins = arduino_leonardo::Pins::new(dp.PORTB, dp.PORTC, dp.PORTD, dp.PORTE);

let mut i2c = arduino_leonardo::I2c::new(
    dp.TWI,
    pins.d2.into_pull_up_input(&mut pins.ddr),
    pins.d3.into_pull_up_input(&mut pins.ddr),
    50000,
);
