[−][src]Type Definition arduino_leonardo::Serial

type Serial<IMODE> = Usart1<MHz16, IMODE>;

Serial (UART) interface on pins D0 (RX) and D1 (TX)

Example

For a full example, see examples/leonardo-serial.rs. In short:

let dp = arduino_leonardo::Peripherals::take().unwrap();

let mut pins = arduino_leonardo::Pins::new(dp.PORTB, dp.PORTC, dp.PORTD, dp.PORTE);

let mut serial = arduino_leonardo::Serial::new(
    dp.USART1,
    pins.d0,
    pins.d1.into_output(&mut pins.ddr),
    57600,
);

ufmt::uwriteln!(&mut serial, "Hello from Arduino!\r").void_unwrap();
